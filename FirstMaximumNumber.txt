package com.company;

public class FindFirstMaxNumber {

    public static void main(String[] args) {

        int arr[]={ 6, 2, 0 ,-3 ,4 ,1 ,7};

        int max=arr[0];
        for(int i=0;i<arr.length;i++){
            if(arr[i]>max){
                max=arr[i];

            }
        }

        System.out.println("The First Maximum Number is : "+ max);
        
    }
}
